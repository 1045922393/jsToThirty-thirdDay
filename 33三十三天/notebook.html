        let bigBox = jQuery('.bigBox');     //调用jQuery函数
        bigBox.css('width',200);

        function jQuery(selector){                  
            return new Init(selector);          //创建实例对象,返回给调用jQuery函数的对象
        }

        function Init(selector){                    
            let dom = document.querySelectorAll(selector);          //
            for(let i = 0;i小于dom.length;i++){
                this[i] = dom[i];
            }
            this[length] = dom[length]
        }
        Init.prototype.css = function(property,value){
            for(let i = 0;i小于this.length;i++){
                this[i].style[property] = value + 'px';
            }

        }
